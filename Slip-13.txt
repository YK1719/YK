package main

import "fmt"

func main() {
    evenSum, oddSum := 0, 0
    for i := 1; i <= 100; i++ {
        if i%2 == 0 {
            evenSum += i
        } else {
            oddSum += i
        }
    }
    fmt.Printf("Sum of Even numbers (1–100): %d\n", evenSum)
    fmt.Printf("Sum of Odd  numbers (1–100): %d\n", oddSum)
}
