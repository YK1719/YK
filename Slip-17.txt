package main

import "fmt"

func arithmetic(a, b float64) (add, sub, mul, div float64) {
    add = a + b
    sub = a - b
    mul = a * b
    if b != 0 { div = a / b }
    return
}

func main() {
    var a, b float64
    fmt.Print("Enter two numbers: ")
    fmt.Scan(&a, &b)

    add, sub, mul, div := arithmetic(a, b)
    fmt.Printf("Add = %.2f\nSub = %.2f\nMul = %.2f\nDiv = %.2f\n", add, sub, mul, div)
}
