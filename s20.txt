Q1. Write a python program to accept a string, Count total number of vowels, consonants and special symbols from it.

# Accept string input
text = input("Enter a string: ")

# Initialize counters
vowels = 0
consonants = 0
specials = 0

# Define vowels
vowel_letters = "aeiouAEIOU"

# Iterate over each character
for char in text:
    if char.isalpha():  # Check if alphabet
        if char in vowel_letters:
            vowels += 1
        else:
            consonants += 1
    elif char.isdigit() or char.isspace():
        continue  # Ignore digits and spaces
    else:
        specials += 1  # Anything else is special symbol

# Display results
print("Total Vowels:", vowels)
print("Total Consonants:", consonants)
print("Total Special Symbols:", specials)

Q2. Write a Python program to accept a file name and a byte position as input. If file exist open a file in read mode and move the file pointer to the given byte position, prints the content of the file from that position till end of file otherwise give appropriate message.

import os

# Accept file name and byte position
file_name = input("Enter file name: ")
byte_pos = int(input("Enter byte position: "))

# Check if file exists
if os.path.exists(file_name):
    try:
        # Open file in read-binary mode to handle byte positions correctly
        with open(file_name, 'rb') as file:
            # Move file pointer to the specified byte position
            file.seek(byte_pos)
            
            # Read the rest of the file
            content = file.read()
            
            # Decode to string assuming utf-8 encoding
            print("\nContent from byte position", byte_pos, ":\n")
            print(content.decode('utf-8'))
    except Exception as e:
        print("Error reading file:", e)
else:
    print("File does not exist.")
