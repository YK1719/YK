Write a python program to accept a string, find all words that do not contain vowels.

s = input("Enter a string: ")
vowels = "aeiouAEIOU"
words = s.split()
print("Words without vowels:")
for word in words:
    has_vowel = False
    for ch in word:
        if ch in vowels:
            has_vowel = True
            break
    if not has_vowel:
        print(word)
